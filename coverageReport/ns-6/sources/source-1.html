42 /**
43  * A watchmen which keeps watching state changes of permissions and events of permissions.
44  */
45 @OptIn(FlowPreview::class)
46 @Suppress("OPT_IN_IS_NOT_ENABLED", "unused")
47 internal class PermissionWatchmen(
48     private val application: Application,
49     dispatcher: CoroutineDispatcher
50 ) {
51     private val watchmenScope = CoroutineScope(
52         dispatcher +
53             SupervisorJob() +
54             CoroutineName("PermissionWatchmen")
55     )
56 
57     private var watchEventsJob: Job? = null
58     private var watchActivityEventJob: Job? = null
59 
60     /**
61      * A in-memory store for storing permission and its state holder i.e. [StateFlow]
62      */
63     private val permissionFlows = mutableMapOf<String, PermissionStateFlowDelegate>()
64 
65     private val permissionEvents = MutableSharedFlow<PermissionEvent>()
66 
67     fun watch(permission: String): StateFlow<PermissionState> {
68         // Wakeup watchmen if sleeping
69         wakeUp()
70         return getOrCreatePermissionStateFlow(permission)
71     }
72 
73     fun watchMultiple(permissions: Array<String>): StateFlow<MultiplePermissionState> {
74         // Wakeup watchmen if sleeping
75         wakeUp()
76 
77         val permissionStates = permissions
78             .distinct()
79             .map { getOrCreatePermissionStateFlow(it) }
80             .toTypedArray()
81 
82         return combineStates(*permissionStates) { MultiplePermissionState(it.toList()) }
83     }
84 
85     fun notifyPermissionsChanged(permissions: Array<String>) {
86         watchmenScope.launch {
87             permissions.forEach { permission ->
88                 permissionEvents.emit(
89                     PermissionEvent(
90                         permission = permission,
91                         isGranted = isPermissionGranted(permission)
92                     )
93                 )
94             }
95         }
96     }
97 
98     @Synchronized
99     fun wakeUp() {
100         watchPermissionEvents()
101         watchActivities()
102         notifyAllPermissionsChanged()
103     }
104 
105     @Synchronized
106     fun sleep() {
107         watchmenScope.coroutineContext.cancelChildren()
108     }
109 
110     /**
111      * First finds for existing flow (if available) otherwise creates a new [MutableStateFlow]
112      * for [permission] and returns a read-only [StateFlow] for a [permission].
113      */
114     @Synchronized
115     private fun getOrCreatePermissionStateFlow(permission: String): StateFlow<PermissionState> {
116         val delegate = permissionFlows[permission] ?: run {
117             val initialState = PermissionState(permission, isPermissionGranted(permission))
118             PermissionStateFlowDelegate(initialState).also { permissionFlows[permission] = it }
119         }
120         return delegate.state
121     }
122 
123     /**
124      * Watches for the permission events and updates appropriate state holders of permission
125      */
126     private fun watchPermissionEvents() {
127         if (watchEventsJob != null && watchEventsJob?.isActive == true) return
128         watchEventsJob = watchmenScope.launch {
129             permissionEvents.collect { (permission, isGranted) ->
130                 permissionFlows[permission]?.setState(PermissionState(permission, isGranted))
131             }
132         }
133     }
134 
135     /**
136      * Watches for activity foreground events (to detect whether user has changed permission by
137      * going in settings) and recalculates state of the permissions which are currently being
138      * observed.
139      */
140     private fun watchActivities() {
141         if (watchActivityEventJob != null && watchActivityEventJob?.isActive == true) return
142         watchActivityEventJob = application.activityForegroundEventFlow
143             // This is just to avoid frequent events.
144             .debounce(DEFAULT_DEBOUNCE_FOR_ACTIVITY_CALLBACK)
145             .onEach {
146                 // Since this is not priority task, we want to yield current thread for other
147                 // tasks for the watchmen.
148                 yield()
149                 notifyAllPermissionsChanged()
150             }
151             .launchIn(watchmenScope)
152     }
153 
154     private fun notifyAllPermissionsChanged() {
155         if (permissionFlows.isEmpty()) return
156         notifyPermissionsChanged(permissionFlows.keys.toTypedArray())
157     }
158 
159     private fun isPermissionGranted(permission: String): Boolean {
160         return ContextCompat.checkSelfPermission(
161             application,
162             permission
163         ) == PackageManager.PERMISSION_GRANTED
164     }
165 
166     /**
167      * A event model for permission event.
168      *
169      * @property permission Name of a permission
170      * @property isGranted State of permission whether it's granted / denied
171      */
172     private data class PermissionEvent(val permission: String, val isGranted: Boolean)
173 
174     /**
175      * A delegate for [MutableStateFlow] which creates flow for holding state of a permission.
176      */
177     private class PermissionStateFlowDelegate(initialState: PermissionState) {
178 
179         private val _state = MutableStateFlow<PermissionState>(initialState)
180         val state = _state.asStateFlow()
181 
182         fun setState(newState: PermissionState) {
183             _state.value = newState
184         }
185     }
